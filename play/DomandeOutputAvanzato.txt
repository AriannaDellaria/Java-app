public class Esercizio1 {
    public static void main(String[] args) {
        Greeter g1 = new Greeter("paolo");
        System.out.print(g1.sayHello());

        Greeter g2 = new Greeter();
        System.out.print(g2.sayHello());

        Greeter g3 = new Greeter("GiOvAnNa");
        g3.setName("maRia");
        System.out.print(g3.sayHello());

        Greeter g4 = new Greeter("");
        System.out.print(g4.sayHello());
    }
}

class Greeter {
    private String name;

    Greeter() {
        this("World");
    }

    Greeter(String name) {
        this.name = formattazioneNome(name);
    }

    public String sayHello() {
        return "Hello, " + this.name + "!";
    }

    public String getName() {
        return this.name;
    }

    public void setName(String name) {
        this.name = formattazioneNome(name);
    }

    private static String formattazioneNome(String name) {
        if (name == null || name.isBlank()) {
            return "World";
        }
        return name.substring(0, 1).toUpperCase() + name.substring(1).toLowerCase();
    }
}

import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;

public class Esercizio2 {
    public static void main(String[] args) {
        try {
            BufferedReader in = new BufferedReader(new InputStreamReader(System.in));
            System.out.println("Quanti anni hai?");
            String inputLine = in.readLine();
            int age = Integer.parseInt(inputLine);
            age++;
            System.out.println("Il prossimo anno avrai " + age + " anni");
        } 
        catch (IOException exception) {
            System.out.println("I/O exception " + exception);
        } 
        catch (NumberFormatException exception) {
            System.out.println("Input was not a number");
        } 
        finally {
            System.out.println("Programma concluso.");
        }
    }
}

import java.io.*;
import java.util.*;

public class Esercizio4{
    public static void main(String[] args) {
        String nomeFile = "dati_meteo.txt";

        try (Scanner scanner = new Scanner(new File(nomeFile))) {
            while (scanner.hasNextLine()) {
                String riga = scanner.nextLine();
                String[] dati = riga.split(", ");
                
                String citta = dati[0];
                double temperatura = Double.parseDouble(dati[1]);
                int umidita = Integer.parseInt(dati[2]);

                System.out.println("Città: " + citta + " | Temperatura: " + temperatura + "°C | Umidità: " + umidita + "%");
            }
        } catch (FileNotFoundException e) {
            System.out.println("Errore: Il file non è stato trovato.");
        }
    }
}

            System.out.println("Quanti studenti vuoi registrare?");
